import org.springframework.boot.CommandLineRunner;
import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.context.annotation.Bean;

@SpringBootApplication
public class Application {

    public static void main(String[] args) {
        SpringApplication.run(Application.class, args);
    }

    @Bean
    CommandLineRunner runner(EmployeeService employeeService) {
        return args -> {
            Employee emp1 = new Employee(1, "John marquee", 50000, 10000, 5000, 2000);
            Employee emp2 = new Employee(2, "Jane Doe", 60000, 12000, 6000, 3000);
            // Add more employees as needed
            employeeService.saveEmployee(emp1);
            employeeService.saveEmployee(emp2);
        };
    }
}
